
# i686

ip link set dev enp0s3 up
dhcpcd enp0s3

useradd -u 1001 -U -G wheel,games,users -c "Guy Yur" guy

# cd to config dir (mounted usb flash, nfs, cifs, ...)
./partition.sh
./create_filesystems.sh
./mount_and_mkdir.sh

nano /etc/pacman.conf
  uncomment CacheDir
  
  CleanMethod = KeepCurrent
  
  change SigLevel to
    SigLevel = Never


mount ? /var/cache/pacman

./pacman.sh

umount /var/cache/pacman

nano /mnt/etc/locale.gen
  en_AU.UTF-8 UTF-8
  #...
  en_GB.UTF-8 UTF-8
  #...
  en_US.UTF-8 UTF-8
  #en_US ISO-8859-1
  #...
  he_IL.UTF-8 UTF-8
  #...

nano /mnt/etc/pacman.conf
  DBPath = /usr/db/pacman/
  
  CleanMethod = KeepCurrent
  
  change SigLevel to
    SigLevel = Never

nano /mnt/etc/pacman.d/mirrorlist
  uncomment lines

./config.sh /mnt

restore users and groups added by pacman packages

chroot /mnt locale-gen

chroot /mnt /usr/bin/mkinitcpio -p linux

./bootloader.sh /mnt

./unmount.sh

systemctl poweroff


config Parcellite
config PCManFM
