  CleanMethod = KeepCurrent
  
  comment CheckSpace
  
  change SigLevel to
    SigLevel = Never

  DBPath = /usr/db/pacman/
  
  CleanMethod = KeepCurrent
  
  change SigLevel to
    SigLevel = Never

  uncomment lines

./config.sh /mnt

restore users and groups added by pacman packages

chroot /mnt locale-gen

./bootloader.sh /mnt
